   # CREATE NETWORK1
   docker network create mynw1
   docker network ls

   #CREATE CONATINER FOR FRONTEND
   docker run -d -it -P --network mynw1 --name myfe nginx:alpine
   docker network inspect mynw1

   #CREATE CONATINER FOR BACKEND
   docker run -d -it --network mynw1 --name mybe php:alpine
   docker network inspect mynw1

   #CREATE NETWORK2
   docker network create mynw2
   #CONNECT NETWORK TO EXISTING CONTAINER
   docker network connect mynw2 mybe

   #CREATE CONATINER FOR DATABASE
   docker run -d -it --network mynw2 --name mydb ubuntu
   docker network inspect mynw2
